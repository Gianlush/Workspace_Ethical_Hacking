Avvio:
	- entra nella cartella "ilBonus"
	- lancia il comando: 
				docker compose up --build -d
	
	
	
Stop:
	- entra nella cartella "ilBonus"
	- lancia il comando: 
				docker compose down -v
	
	(l'opzione -v serve per eliminare i volumi utilizzati così da non riempire la memoria del PC velocemente)
